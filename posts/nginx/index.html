安装

下载gz文件，解压并编译

1
2
3
4

[root@nginx ~]# ls
anaconda-ks.cfg  nginx-1.20.1.tar.gz
[root@nginx ~]# tar -zxf nginx-1.20.1.tar.gz 
[root@nginx nginx-1.20.1]# ./configure --prefix=/usr/local/nginx

发现执行configure错误，在centos7.6下：

1
2
3
4

./configure: error: the HTTP rewrite module requires the PCRE library.
You can either disable the module by using --without-http_rewrite_module
option, or install the PCRE library into the system, or build the PCRE library
statically from the source with nginx by using --with-pcre=<path> option.

提示缺少PCRE的库，这是个perl的库，安装一下：

1

[root@nginx nginx-1.20.1]# yum install -y pcre pcre-devel

重新执行：

1

[root@nginx nginx-1.20.1]# ./configure --prefix=/usr/local/nginx

出现错误：

1
2
3
4

./configure: error: the HTTP gzip module requires the zlib library.
You can either disable the module by using --without-http_gzip_module
option, or install the zlib library into the system, or build the zlib library
statically from the source with nginx by using --with-zlib=<path> option.

需要安装库gzip:

1

[root@nginx nginx-1.20.1]# yum install -y zlib zlib-devel

再次执行./configure:

1

[root@nginx nginx-1.20.1]# ./configure --prefix=/usr/local/nginx

执行成功。

执行make安装：

1
2

[root@nginx nginx-1.20.1]# make
[root@nginx nginx-1.20.1]# make install

进入安装好的目录，执行nginx：

1
2
3
4

[root@nginx sbin]# cd /usr/local/nginx/sbin/
[root@nginx sbin]# ls
nginx
[root@nginx sbin]# ./nginx

打开网页，确认服务器防火墙关闭或者打开端口：

1
2

[root@nginx sbin]#  firewall-cmd --zone=public --add-port=80/tcp --permanent
[root@nginx sbin]#  firewall-cmd --reload

创建service文件，使nginx服务随机器启动：

[Unit]
Description=nginx -  web server
After=network.target remote-fs.target nss-lookup.target
  
[Service]
Type=forking
PIDFile=/usr/local/nginx/logs/nginx.pid
ExecStartPre=/usr/local/nginx/sbin/nginx -t -c /usr/local/nginx/conf/nginx.conf
ExecStart=/usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf
ExecReload=/usr/local/nginx/sbin/nginx -s reload
ExecStop=/usr/local/nginx/sbin/nginx -s stop
ExecQuit=/usr/local/nginx/sbin/nginx -s quit
PrivateTmp=true
  
[Install]
WantedBy=multi-user.target

首先停掉现有进程,然后重新加载服务，启动nginx服务：

1
2
3

[root@nginx sbin]# ./nginx -s stop
[root@nginx sbin]# systemctl daemon-reload
[root@nginx sbin]# systemctl start nginx.service

反向代理负载均衡

nginx可以实现反向代理的负载均衡，这里我们使用nginx在同一台机器上做实验，启动三个nginx，一种一台通过80端口负载，后面两个nginx分别为端口88，和 99

  • 编写88端口和99端口的nginx配置文件，使他们分别指向不同的网页

    1
2
3
4
5
6
7

    [root@nginx conf]# cp nginx.conf nginx-88.conf
[root@nginx conf]# cp nginx.conf nginx-99.conf
[root@nginx conf]# mkdir /html/88 -p
[root@nginx conf]# mkdir /html/99 -p
[root@nginx conf]# echo "I'm from 88 port." > /html/88/index.html
[root@nginx conf]# echo "I'm from 99 port." > /html/99/index.html

  • 编写80端口的nginx配置文件，通过代理负载到88和99端口的nginx

    nginx.conf:

    upstream zkun {
server 192.168.149.28:88 weight=1;
server 192.168.149.28:99 weight=10;
}
server {
    listen       80;
    server_name  localhost;

    #charset koi8-r;

    #access_log  logs/host.access.log  main;

    location / {
        proxy_pass http://zkun;
 }

    重新启动三个nginx，先启动系统服务的哪个，避免重启会干掉后面2个nginx：

    1
2
3
4

    root@nginx sbin]# systemctl restart nginx.service
[root@nginx sbin]# ./nginx -c conf/nginx-88.conf
[root@nginx sbin]# ./nginx -c conf/nginx-99.conf

    其中upstream里的weight表示权重

    https 配置

    查看nginx是否有ssl模块，如果没有需要重新编译

    1

    [root@nginx nginx-1.20.1]# ./nginx -V

    结果信息中如果没有--with-http_ssl_module信息，则需要重新编译：

    1

    [root@nginx nginx-1.20.1]# ./configure --with-http_stub_status_module --with-http_ssl_module
